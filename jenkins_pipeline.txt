pipeline{
    agent any
    tools{
        jdk 'jdk17'
        nodejs 'node16'
    }
    environment {
        SCANNER_HOME=tool 'sonar-scanner'
    }
    stages {
        stage('clean workspace'){
            steps{
                cleanWs()
            }
        }
        stage('Checkout from Git'){
            steps{
                git credentialsId: 'f1447528-919f-41bb-8ee3-0a1778b85f46' , branch: 'main', url: 'https://github.com/faroukrajhi/Netflix-DevSecOps.git'
            }
        }
        stage("Sonarqube Analysis "){
            steps{
                withSonarQubeEnv('sonar-server') {
                    sh ''' $SCANNER_HOME/bin/sonar-scanner -Dsonar.projectName=Netflix \
                    -Dsonar.projectKey=Netflix '''
                }
            }
        }
        
        stage('Install Dependencies') {
            steps {
                sh "npm install"
            }
        }
        
        stage("Docker Build & Push"){
            steps{
                script{
                   withDockerRegistry(credentialsId: 'a50c698c-ff32-4319-a393-abc15e71fd75', toolName: 'docker'){   
                       sh "docker build --build-arg TMDB_V3_API_KEY=2d5c85fe0603ad63040738f88228a14f -t faroukrajhi/netflix:v${BUILD_NUMBER} ."
                       sh "docker push faroukrajhi/netflix:v${BUILD_NUMBER} "
                    }
                }
            }
        }
        stage("TRIVY"){
            steps{
                sh "trivy image faroukrajhi/netflix:latest > trivyimage.txt" 
            }
        }
        stage('Update Deployment YAML and Push to GitHub') {
            steps {
                script {
                    dir('k8s') {
                        // Update the image tag dynamically
                        sh """
                        sed -i 's|image: faroukrajhi/netflix:.*|image: faroukrajhi/netflix:v${BUILD_NUMBER}|' deployment.yml
                        """
        
                        // Configure Git (set your name and email for commit)
                        sh '''
                        git config user.email "jenkins@example.com"
                        git config user.name "Jenkins"
                        '''
        
                        // Add, commit, and push the changes using GitHub credentials
                        withCredentials([usernamePassword(credentialsId: 'f1447528-919f-41bb-8ee3-0a1778b85f46', usernameVariable: 'GIT_USERNAME', passwordVariable: 'GIT_PASSWORD')]) {
                            sh '''
                            git add deployment.yml
                            git commit -m "Update deployment image to build v${BUILD_NUMBER}"
                            git push https://${GIT_USERNAME}:${GIT_PASSWORD}@github.com/faroukrajhi/Netflix-DevSecOps.git HEAD:main
                            '''
                        }
                    }
                }
            }
        }

    }
}
